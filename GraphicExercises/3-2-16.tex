Which of the following sets of vectors, drawn as stereo pairs, appear to form an orthogonal set?  
Which appear to form an orthonormal set?

\def\threevcol{blue}
\newcommand{\threev}[5][below]{
  \addplot3[gray] coordinates 
  {(#2,#3,#4)(#2,#3,0)(#2,0,0)(0,0,0)(0,#3,0)(#2,#3,0)};
  \addplot3[\threevcol,thick,quiver={u=#2,v=#3,w=#4},-stealth] 
  coordinates {(0,0,0)};
  \node[#1] at (axis cs:#2,#3,#4) {$#5$};
  }
\newcommand{\ThreeVec}[9]{\qview{35}{39}{
\begin{tikzpicture} 
    \begin{axis}[axis equal, footnotesize, height=5cm, view={\q}{30}]
        \threev{#1}{#2}{#3}{}
        \threev{#4}{#5}{#6}{}
        \threev{#7}{#8}{#9}{}
    \end{axis}
\end{tikzpicture}}}

\ThreeVec{-0.84}{0.33}{0.43}{-0.45}{-0.86}{-0.23}{0.29}{-0.38}{0.88}
\answer{Orthonormal.}
